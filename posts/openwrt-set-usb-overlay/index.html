## OpenWrt设置usb扩容

原文

# 连接SSH

# 安装软件包

opkg update
opkg install block-mount kmod-fs-ext4 e2fsprogs parted kmod-usb-storage

# 查看USB设备位置

ls -l /sys/block

如果是无线路由器只有一个外接存储，大概率是sda

lrwxrwxrwx    1 root     root             0 xxx xx xx:xx sda -> ../devices/xxx/block/sda

# 在USB上创建分区，格式化，设置exroot并自动挂载

#Partitioning and formatting
DISK="/dev/sda"
parted -s ${DISK} -- mklabel gpt mkpart extroot 2048s -2048s
DEVICE="${DISK}1"
mkfs.ext4 -L extroot ${DEVICE}

#Configuring extroot
eval $(block info ${DEVICE} | grep -o -e 'UUID="\S*"')
eval $(block info | grep -o -e 'MOUNT="\S*/overlay"')
uci -q delete fstab.extroot
uci set fstab.extroot="mount"
uci set fstab.extroot.uuid="${UUID}"
uci set fstab.extroot.target="${MOUNT}"
uci commit fstab

#Configuring rootfs_data / ubifs
ORIG="$(block info | sed -n -e '/MOUNT="\S*\/overlay"/s/:\s.*$//p')"
uci -q delete fstab.rwm
uci set fstab.rwm="mount"
uci set fstab.rwm.device="${ORIG}"
uci set fstab.rwm.target="/rwm"
uci commit fstab

按y，一路回车

# 复制内置闪存的文件

mount ${DEVICE} /mnt
tar -C ${MOUNT} -cvf - . | tar -C /mnt -xf -

此时检查software的可用容量应该是U盘的容量了

  • 实测直接拔掉U盘重启就会自动使用内置的闪存启动,不需要任何配置
  • 安装系统更新不确定会发生什么,最好不要这么做
