## HTTP3(quic) 直连黑名单域名测试

根据揭示并绕过中国防火长城基于SNI的QUIC封锁机制,我的看法:
目前(2025年8月)

  • 对quic的封锁漏洞百出,可以轻松绕过
  • 虽然少量流量瘫痪gfw是gfw的bug可以被修复,但审查quic连接的计算量还是比http1/2大不少,这是quic特性无法解决
  • gfw审查不过来quic的时候会直接放行
  • 虽然src_port < dst_port时gfw会直接放行但使用< 443的端口需要提权,不合适

# http3中国直连黑名单域名测试方法

curl测试 (ubuntu curl 8.5.0)

curl --connect-to example.com:443:[x:x:x:x::x]:443 https://example.com

quic测试 (quic-go 0.54.0)

./quic-dl --connect-to [x:x:x:x::x]:443 https://example.com index.html
  • 所有网站使用dns.google取得无污染解析,用无污染的ipv6直连
  • 套Fastly CDN的网站都使用www.fastly.com解析的ipv6地址连接(除twitch.tv报错),所以pinterest.comdns查不到ipv6也是用ipv6连的
  • quic建立连接前先用本机同端口发送两个80-120 bytes的随机数据包干扰审查
  • 不确定这些quic成功连接的网站是
    1.不在quic sni的黑名单里
    2.通过先发送两个随机数据包绕过了gfw的sni审查
    3.gfw负载太大直接放行

# quic-dl

go.mod

module quic-dl

go 1.25

require (
    github.com/quic-go/quic-go v0.54.0
)

main.go

package main

import (
	"context"
	"crypto/tls"
	"flag"
	"fmt"
	"io"
	"log"
	"math/rand"
	"net"
	"net/http"
	"os"
	"strings"

	quic "github.com/quic-go/quic-go"
	"github.com/quic-go/quic-go/http3"
)

func sendRandomUDP(conn *net.UDPConn, dst *net.UDPAddr) (int, error) {
	// Determine random length in [80, 120]
	length := rand.Intn(41) + 80

	// Fill buffer with cryptographically random bytes
	buf := make([]byte, length)
	if _, err := rand.Read(buf); err != nil {
		return 0, fmt.Errorf("failed to generate random data: %w", err)
	}

	// Send via the shared UDPConn
	n, err := conn.WriteToUDP(buf, dst)
	if err != nil {
		return n, fmt.Errorf("failed to send data: %w", err)
	}
	return n, nil
}

func main() {
	fmt.Println("Run")
	// Optional flags
	method := flag.String("X", "GET", "HTTP method to use")
	headers := flag.String("H", "", "HTTP headers, separated by \\n")
	data := flag.String("d", "", "Request body data")
	connectTo := flag.String("connect-to", "", "Override QUIC dial target, e.g. 1.1.1.1:443 or [::1]:443")
	flag.Parse()

	if flag.NArg() != 2 {
		flag.Usage()
		os.Exit(1)
	}

	url := flag.Arg(0)
	outPath := flag.Arg(1)

	// Build HTTP request
	var body io.Reader
	if *data != "" {
		body = strings.NewReader(*data)
	}
	req, err := http.NewRequest(*method, url, body)
	if err != nil {
		log.Fatalf("creating request: %v", err)
	}
	if *headers != "" {
		for _, h := range strings.Split(*headers, "\n") {
			parts := strings.SplitN(h, ":", 2)
			if len(parts) != 2 {
				log.Fatalf("invalid header: %s", h)
			}
			req.Header.Add(strings.TrimSpace(parts[0]), strings.TrimSpace(parts[1]))
		}
	}

	// Prepare HTTP/3 transport
	quicConfig := &quic.Config{}
	h3 := &http3.Transport{
		TLSClientConfig: &tls.Config{
			InsecureSkipVerify: true,
			NextProtos:         []string{"h3"},
		},
		QUICConfig: quicConfig,
	}

	// If --connect-to is set, override Dial
	if *connectTo != "" {
		remoteAddr, err := net.ResolveUDPAddr("udp", *connectTo)
		if err != nil {
			log.Fatalf("resolving %s: %v", *connectTo, err)
		}
		pktConn, err := net.ListenUDP("udp", nil)
		
		localAddr := pktConn.LocalAddr()
		fmt.Printf("Using local address: %v\n", localAddr)
		if err != nil {
			log.Fatalf("opening UDP socket: %v", err)
		}

		h3.Dial = func(ctx context.Context, _ string, tlsCfg *tls.Config, cfg *quic.Config) (*quic.Conn, error) {
			return quic.DialEarly(ctx, pktConn, remoteAddr, tlsCfg, cfg)
		}

		sendRandomUDP(pktConn, remoteAddr)

		sendRandomUDP(pktConn, remoteAddr)

	}

	client := &http.Client{Transport: h3}
	defer h3.Close()

	// Execute request
	resp, err := client.Do(req)
	if err != nil {
		log.Fatalf("HTTP/3 request failed: %v", err)
	}
	defer resp.Body.Close()

	if resp.ProtoMajor != 3 {
		log.Fatalf("server does not support HTTP/3: %s", resp.Proto)
	}

	// Save response to file
	outFile, err := os.Create(outPath)
	if err != nil {
		log.Fatalf("create file: %v", err)
	}
	defer outFile.Close()

	if _, err := io.Copy(outFile, resp.Body); err != nil {
		log.Fatalf("writing file: %v", err)
	}

	if *connectTo != "" {
		fmt.Printf("Downloaded %s to %s via HTTP/3 (connect-to=%s)\n", url, outPath, *connectTo)
	} else {
		fmt.Printf("Downloaded %s to %s via HTTP/3 (default dial)\n", url, outPath)
	}
}

build & run

go mod tidy
go run main.go
